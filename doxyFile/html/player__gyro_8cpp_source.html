154  // Read the raw values.
155  // Read 4 bytes at once,
156  // containing x_gyro (2 bytes) and y_gyro (2 bytes).
157  // With the default settings of the MPU-6050,
158  // there is no filter enabled, and the values
159  // are not very stable. But they are fine for left-right,up-down
160  // determination.
161  gyro_error_check = MPU6050_read(
162  MPU6050_ACCEL_XOUT_H, (uint8_t *)&accel_t_gyro, sizeof(accel_t_gyro));
163 
164  // uncomment code below for debugging purposes.
165  /* Serial.print(F("Read accel, temp and gyro, gyro_error_check = "));
166 Serial.println(gyro_error_check,DEC);
167 */
168 
169  // swap low and high byte of x_gyro and y_gyro
170  swap(&accel_t_gyro.reg.x_gyro_h, &accel_t_gyro.reg.x_gyro_l);
171  swap(&accel_t_gyro.reg.y_gyro_h, &accel_t_gyro.reg.y_gyro_l);
172 
173  // Determine direction
174  // Compare x_gyro and y_gyro value to a THRESHOLD number - it can be
175  // calibrated for the sensitivity process x_gyro
176  if (accel_t_gyro.value.x_gyro < -THRESHOLD) {
177  Serial.print(F("RIGHT \t"));
178  player_package.right = 1;
179  } else if (accel_t_gyro.value.x_gyro > THRESHOLD) {
180  Serial.print(F("LEFT \t"));
181  player_package.left = 1;
182  } else {
183  Serial.print(F(" \t"));
184  // player_package.packet_data = player_package.packet_data & 0xFFE7 ;//
185  // 11111111 11100111;
186  player_package.left = 0;
187  player_package.right = 0;
188  }
189  // process y_gyro
190  if (accel_t_gyro.value.y_gyro < -THRESHOLD) {
191  Serial.print(F("UP \t"));
192  player_package.up = 1;
193  } else if (accel_t_gyro.value.y_gyro > THRESHOLD) {
194  Serial.print(F("DOWN \t"));
195  player_package.down = 1;
196  } else {
197  Serial.print(F(" \t"));
198  player_package.down = 0;
199  player_package.up = 0;
200  }
